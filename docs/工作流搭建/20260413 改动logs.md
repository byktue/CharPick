```bash
Set-Location 'e:\codes\CharPick'
conda activate charpick
@'
from pathlib import Path
import json
from backend.llm_chapter_extraction import extract_chapter_structured

chapter_dir = Path('local_cache/test_sf_test_001/chapter_content_text')
out_dir = chapter_dir.parent / 'extraction_json'
out_dir.mkdir(parents=True, exist_ok=True)

files = sorted(chapter_dir.glob('sf_test_001_chapter_*.txt'))
for file_path in files:
    text = file_path.read_text(encoding='utf-8')
    lines = [line.strip() for line in text.splitlines() if line.strip()]
    chapter_title = lines[0] if lines else file_path.stem
    data = extract_chapter_structured(chapter_title=chapter_title, chapter_text=text)
    (out_dir / f'{file_path.stem}.json').write_text(
        json.dumps(data, ensure_ascii=False, indent=2),
        encoding='utf-8'
    )

print(json.dumps({
    'chapter_count': len(files),
    'extraction_dir': str(out_dir)
}, ensure_ascii=False, indent=2))
```
